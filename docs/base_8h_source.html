481 static inline int ccs_datum_cmp(ccs_datum_t a, ccs_datum_t b) {
482  if (a.type < b.type) {
483  return -1;
484  } else if (a.type > b.type) {
485  return 1;
486  } else {
487  switch(a.type) {
488  case CCS_STRING:
489  if (a.value.s == b.value.s)
490  return 0;
491  else if (!a.value.s)
492  return -1;
493  else if (!b.value.s)
494  return 1;
495  else
496  return strcmp(a.value.s, b.value.s);
497  break;
498  case CCS_INTEGER:
499  return a.value.i < b.value.i ? -1 :
500  a.value.i > b.value.i ? 1 : 0;
501  break;
502  case CCS_FLOAT:
503  return a.value.f < b.value.f ? -1 :
504  a.value.f > b.value.f ? 1 : 0;
505  break;
506  case CCS_NONE:
507  case CCS_INACTIVE:
508  return 0;
509  break;
510  default:
511  return memcmp(&(a.value), &(b.value), sizeof(ccs_value_t));
512  }
513  }
