593 static inline int
594 ccs_datum_cmp(ccs_datum_t a, ccs_datum_t b)
595 {
596  if (a.type < b.type) {
597  return -1;
598  } else if (a.type > b.type) {
599  return 1;
600  } else {
601  switch (a.type) {
602  case CCS_DATA_TYPE_STRING:
603  if (a.value.s == b.value.s)
604  return 0;
605  else if (!a.value.s)
606  return -1;
607  else if (!b.value.s)
608  return 1;
609  else
610  return strcmp(a.value.s, b.value.s);
611  break;
612  case CCS_DATA_TYPE_INT:
613  return a.value.i < b.value.i ? -1 :
614  a.value.i > b.value.i ? 1 :
615  0;
616  break;
617  case CCS_DATA_TYPE_FLOAT:
618  return a.value.f < b.value.f ? -1 :
619  a.value.f > b.value.f ? 1 :
620  0;
621  break;
622  case CCS_DATA_TYPE_NONE:
623  case CCS_DATA_TYPE_INACTIVE:
624  return 0;
625  break;
626  default:
627  return memcmp(
628  &(a.value), &(b.value), sizeof(ccs_value_t));
629  }
630  }
631 }
